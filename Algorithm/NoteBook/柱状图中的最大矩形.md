```cpp
int largestRectangleArea(vector<int>& heights) {
        int n = heights.size();
        vector<int> tmp(n);
        tmp = heights;
        sort(tmp.begin(),tmp.end());

        int maxS = 0;
        for(int i = 0;i < n;i ++){
            int curH = tmp[i];
            int count = 0;
            int curMax = 0;
            for(int j = 0;j < n;j ++){
                if(heights[j] >= curH){
                    count += curH;
                    curMax = max(curMax,count);
                }else{
                    count = 0;
                }
            }
            maxS = max(maxS,curMax);
        }
        return maxS;
    } 
```

```cpp
int getMaxS(vector<int>& heights,int i,int curS){
        int left = 0, right = 0;
        for(int j = i - 1;j >= 0;j --){
            if(heights[j] < heights[i]) break;
            left += heights[i];
        }
        for(int j = i + 1;j < heights.size();j ++){
            if(heights[j] < heights[i]) break;
            right += heights[i];
```

第二次优化：单调栈O(n)
```cpp
    int largestRectangleArea(vector<int>& heights) {
        int n = heights.size();
        vector<int> left(n,0);
        vector<int> right(n,n - 1);
        
        vector<int> stk;
        stk.push_back(0);
        for(int i = 1;i < n;i ++){
            while(!stk.empty() && heights[i] < heights[stk.back()]){
                right[stk.back()] = i - 1;
                stk.pop_back();
            }
            stk.push_back(i);
        }
        stk.clear();
        stk.push_back(n - 1);
        for(int i = n - 2;i >= 0;i --){
            while(!stk.empty() && heights[i] < heights[stk.back()]){
                left[stk.back()] = i + 1;
                stk.pop_back();
            }
            stk.push_back(i);
        }

        int maxS = 0;
        for(int i = 0;i < n;i ++){
            maxS = max(maxS,heights[i] * (right[i] - left[i] + 1));
        }
        return maxS;
    }
```

第三次：同阶内时间优化
```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int n = heights.size();
        vector<int> left(n,0);
        vector<int> right(n,n - 1);
        vector<int> stk;
        stk.push_back(0);
        for(int i = 1;i < n;i ++){
            while(!stk.empty() && heights[i] < heights[stk.back()]){
                right[stk.back()] = i - 1;
                stk.pop_back();
            }
            if(!stk.empty()) left[i] = stk.back() + 1;
            stk.push_back(i);
        }

        int maxS = 0;
        for(int i = 0;i < n;i ++){
            maxS = max(maxS,heights[i] * (right[i] - left[i] + 1));
        }
        return maxS;
    } 
};
```
