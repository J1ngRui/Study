标准二分
```cpp
    int searchL(vector<int>& nums,int target){
        int ans = -1;
        int l = 0,r = nums.size() - 1;
        while(l <= r){
            int mid = (l + r) / 2;
            if(target == nums[mid]) 
                return mid;
            else if(target < nums[mid]){
                r = mid - 1;
            }else{
                l = mid + 1;
            }
        }
        return ans;
    }
```

二分查找(左)
```cpp
int searchL(vector<int>& nums,int target){
        int ans = -1;
        int l = 0,r = nums.size() - 1;
        while(l <= r){
            int mid = (l + r) / 2;
            if(target <= nums[mid]){
                if(target == nums[mid]) ans = mid;
                r = mid - 1;
            }else{
                // nums[mid] < target
                l = mid + 1;
            }
        }
        return ans;
    }
```
二分查找(右)
```cpp
int searchR(vector<int>& nums,int target){
        int ans = -1;
        int l = 0,r = nums.size() - 1;
        while(l <= r){
            int mid = (l + r + 1) / 2;
            if(target < nums[mid]){
                r = mid - 1;
            }else{
                // nums[mid] < target
                if(target == nums[mid]) ans = mid;
                l = mid + 1;
            }
        }
        return ans;
    }
```
